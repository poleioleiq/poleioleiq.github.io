using namespace std;
const int maxn = 1e4 + 5;
int ran[maxn];
int pre[maxn];

int n, m;

void init()
{
for (int i = 1; i <= n; i++) {
pre[i] = i;
ran[i] = 1;
}
}

int find(int x)
{
if (x == pre[x]) return x;
return pre[x] = find(pre[x]);
}

void join(int x, int y)
{
int fx = find(x);
int fy = find(y);

if (fx == fy) return;
if (ran[x] < ran[y]) pre[fx] = fy;
else {
    pre[fy] = fx;
    if (ran[fx] == ran[fy]) ran[fx]++;
}

}

int main()
{
FAST_IO;
while (cin >> n, n) {
cin >> m;
init();
int a, b;
while (m–) {
cin >> a >> b;
join(a, b);
}
int ans = 0;
for (int i = 1; i <= n; i++) {
if (pre[i] == i) ans++;
}
cout << ans - 1 << endl;
}
return 0;
}
